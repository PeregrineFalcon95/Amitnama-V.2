10004 - Bicoloring (Uva) Online Judge code in C++

#include<bits/stdc++.h>
using namespace std;
int n , m;
bool bfs();
int main()
{
    ios_base::sync_with_stdio(NULL);
    cin.tie(NULL);
    cout.tie(NULL);
    while( cin>>n && n )
    {
        cin>>m;
        if ( bfs() )cout<<"NOT BICOLORABLE.\n";
        else cout<<"BICOLORABLE.\n";
    }
    return 0;
}
bool bfs()
{
    vector < int > graph [ n + 1 ];
    int level [ n + 1 ];

    ///Marking the colors as not colored with 'n'
    memset( level , 0 , sizeof level );

    int a , b;
    while( m-- ) ///Taking input of m edges
    {
        cin>>a>>b;
        graph [ a ].push_back( b );
        graph [ b ].push_back( a );
    }

    queue < int > qu;
    qu.push( 0 ); ///Pushing source
    level [ 0 ] = 0; ///Marking color with white

    ///Checking bi-coloring through bfs
    while( !qu.empty() )
    {
        a = qu.front();
        qu.pop();
        int sz = graph [ a ].size();
        for ( int i = 0 ; i < sz; i++ )
        {
            b = graph [ a ] [ i ];
            if ( !level [ b ] )
            {
                qu.push( b );
                level [ b ] = level [ a ] + 1;
            }
            else if ( level [ b ] == level [ a ] )return 1;
        }
    }
    return 0;
}
